552 VectorXd PathPlanner::solveqp(QP_form qp_prob,bool& is_ok){
553  is_ok = true;
554  MatrixXd Q = qp_prob.Q;
555  MatrixXd H = qp_prob.H;
556  MatrixXd Aineq = qp_prob.A;
557  MatrixXd bineq = qp_prob.b;
558  MatrixXd Aeq = qp_prob.Aeq;
559  MatrixXd beq = qp_prob.beq;
560  if(is_this_verbose){
561  cout<<"Q: "<<endl;
562  cout<<Q<<endl;
563 
564  cout<<"H: "<<endl;
565  cout<<H<<endl;
566 
567  cout<<"Aineq: "<<endl;
568  cout<<Aineq<<endl;
569 
570  cout<<"bineq: "<<endl;
571  cout<<bineq<<endl;
572 
573  cout<<"Aeq: "<<endl;
574  cout<<Aeq<<endl;
575 
576  cout<<"beq: "<<endl;
577  cout<<beq<<endl;
578  }
579  USING_NAMESPACE_QPOASES;
580 
581  int N_var = Q.rows();
582  int N_ineq_const = Aineq.rows();
583  int N_eq_const = Aeq.rows();
584  int N_const = N_ineq_const + N_eq_const;
585 
586  real_t H_qp[N_var*N_var];
587  real_t g[N_var];
588  real_t A[N_var*N_const];
589  real_t lbA[N_const];
590  real_t ubA[N_const];
591 
592  // cost
593  for (int i = 0;i<N_var;i++){
594  g[i] = H(0,i);
595  for(int j = 0;j<N_var;j++)
596  H_qp[j*N_var+i] = 2*Q(i,j);
597  }
598 
599  // eq constraints
600  for (int r = 0; r<N_eq_const;r++){
601  lbA[r] = beq(r);
602  ubA[r] = beq(r);
603  for(int c= 0; c<N_var;c++)
604  A[r*N_var + c] = Aeq(r,c);
605  }
606 
607  // ineq constraints
608  for (int r = 0; r<N_ineq_const;r++){
609  lbA[N_eq_const+r] = -1e+8;
610  ubA[N_eq_const+r] = bineq(r);
611  for(int c= 0; c<N_var;c++)
612  A[(r+N_eq_const)*N_var + c] = Aineq(r,c);
613  }
614 
615  int_t nWSR = 2000;
616 
617  QProblem qp_obj(N_var,N_const,HST_SEMIDEF);
618  std::cout<<"hessian type: "<<qp_obj.getHessianType()<<endl;
619  Options options;
620  options.printLevel = PL_LOW;
621  qp_obj.setOptions(options);
622  qp_obj.init(H_qp,g,A,NULL,NULL,lbA,ubA,nWSR);
623  if(qp_obj.isInfeasible()){
624  cout<<"[QP solver] warning: problem is infeasible. "<<endl;
625  is_ok = false;
626  }
627  real_t xOpt[N_var];
628  qp_obj.getPrimalSolution(xOpt);
629 
630  if(not qp_obj.isSolved()){
631  cout<<"[QP solver] quadratic programming has not been solved "<<endl;
632  is_ok = false;
633  }
634 
635  VectorXd sol(N_var);
636 
637  for(int n = 0; n<N_var;n++)
638  sol(n) = xOpt[n];
639  return sol;
640 }
